ZADANIE 2

docker run -it --name notificationservice -p 92:80 mcr.microsoft.com/dotnet/sdk:8.0

w drugiej konsoli:

docker cp Library.NotificationService2 notificationservice:/app

w pierwszej konsoli:

cd app/
dotnet restore
dotnet publish -c Release -o out
cd out/
dotnet Library.NotificationService2.dll

w drugiej:

docker commit notificationservice notificationserviceimg
docker stop notificationservice
docker rm notificationservice
